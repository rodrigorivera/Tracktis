022    package groovyx.net.http;
023    
024    import java.io.IOException;
025    import java.io.InputStream;
026    import java.net.URI;
027    import java.net.URL;
028    import java.security.GeneralSecurityException;
029    import java.security.KeyStore;
030    
031    import org.apache.http.auth.AuthScope;
032    import org.apache.http.auth.UsernamePasswordCredentials;
033    import org.apache.http.conn.scheme.Scheme;
034    import org.apache.http.conn.ssl.SSLSocketFactory;
035    
036    /**
037     * Encapsulates all configuration related to HTTP authentication methods.
038     * @see HTTPBuilder#getAuth()
039     * 
040     * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
041     */
042    public class AuthConfig {
043            protected HTTPBuilder builder;
044            public AuthConfig( HTTPBuilder builder ) {
045                    this.builder = builder;
046            }
047            
048            /**
049             * Set authentication credentials to be used for the current 
050             * {@link HTTPBuilder#getUri() default host}.  This method name is a bit of 
051             * a misnomer, since these credentials will actually work for "digest" 
052             * authentication as well.
053             * @param user
054             * @param pass
055             */
056            public void basic( String user, String pass ) {
057                    URI uri = ((URIBuilder)builder.getUri()).toURI();
058                    if ( uri == null ) throw new IllegalStateException( "a default URI must be set" );
059                    this.basic( uri.getHost(), uri.getPort(), user, pass );
060            }
061            
062            /**
063             * Set authentication credentials to be used for the given host and port. 
064             * @param host
065             * @param port
066             * @param user
067             * @param pass
068             */
069            public void basic( String host, int port, String user, String pass ) {
070                    builder.getClient().getCredentialsProvider().setCredentials( 
071                            new AuthScope( host, port ),
072                            new UsernamePasswordCredentials( user, pass )
073                    );
074            }
075            
076            /**
077             * Sets a certificate to be used for SSL authentication.  See 
078             * {@link Class#getResource(String)} for how to get a URL from a resource 
079             * on the classpath.
080             * @param certURL URL to a JKS keystore where the certificate is stored.  
081             * @param password password to decrypt the keystore
082             */
083            public void certificate( String certURL, String password ) 
084                            throws GeneralSecurityException, IOException {
085                    
086                    KeyStore keyStore = KeyStore.getInstance( KeyStore.getDefaultType() );
087            InputStream jksStream = new URL(certURL).openStream();
088            try {
089                    keyStore.load( jksStream, password.toCharArray() );
090            } finally { jksStream.close(); }
091    
092            SSLSocketFactory ssl = new SSLSocketFactory(keyStore, password);
093            ssl.setHostnameVerifier( SSLSocketFactory.STRICT_HOSTNAME_VERIFIER );
094            
095            builder.getClient().getConnectionManager().getSchemeRegistry()
096                    .register( new Scheme("https", ssl, 443) );
097            }
098    }
