022    package groovyx.net.http;
023    
024    import static groovyx.net.http.ContentEncoding.Type.DEFLATE;
025    
026    import java.io.IOException;
027    import java.io.InputStream;
028    import java.util.zip.InflaterInputStream;
029    
030    import org.apache.http.HttpEntity;
031    import org.apache.http.entity.HttpEntityWrapper;
032    
033    /**
034     * Content encoding used to handle Deflate responses.
035     * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
036     */
037    public class DeflateEncoding extends ContentEncoding {
038            
039            /**
040             * Returns the {@link ContentEncoding.Type#DEFLATE} encoding string which is 
041             * added to the <code>Accept-Encoding</code> header by the base class.
042             */
043            @Override
044            public String getContentEncoding() {
045                    return DEFLATE.toString();
046            }
047            
048            
049            /**
050             * Wraps the raw entity in a {@link InflaterEntity}.
051             */
052            @Override
053            public HttpEntity wrapResponseEntity( HttpEntity raw ) {
054                    return new InflaterEntity( raw );
055            }
056    
057            /**
058             * Entity used to interpret a Deflate-encoded response
059             * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
060             */
061        public static class InflaterEntity extends HttpEntityWrapper {
062    
063            public InflaterEntity(final HttpEntity entity) {
064                super(entity);
065            }
066        
067            /**
068             * returns a {@link InflaterInputStream} which wraps the original entity's
069             * content stream
070             * @see HttpEntity#getContent()
071             */
072            @Override
073            public InputStream getContent() throws IOException, IllegalStateException {
074                return new InflaterInputStream( wrappedEntity.getContent() );
075            }
076    
077            /**
078             * @return -1
079             */
080            @Override
081            public long getContentLength() {
082                // length of ungzipped content is not known
083                return -1;
084            }
085        }
086    
087    }
