022    package groovyx.net.http;
023    
024    /**
025     * Wraps an error response in an exception for flow control purposes.  That is,
026     * you can still inspect response headers, but in a 
027     * <code>catch( HttpResponseException ex ) {  }</code> block. 
028     * 
029     * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
030     * @since 0.5
031     */
032    public class HttpResponseException extends org.apache.http.client.HttpResponseException {
033            
034            private static final long serialVersionUID = -34809347677236L;
035    
036            HttpResponseDecorator response;
037            
038            public HttpResponseException( HttpResponseDecorator resp ) {
039                    super( resp.getStatusLine().getStatusCode(), 
040                                    resp.getStatusLine().getReasonPhrase() );
041                    this.response = resp;
042            }
043            
044            public HttpResponseDecorator getResponse() {
045                    return response;
046            }
047    }
