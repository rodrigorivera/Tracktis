57  /** <p>
58   * Groovy DSL for easily making HTTP requests, and handling request and response
59   * data.  This class adds a number of convenience mechanisms built on top of 
60   * Apache HTTPClient for things like URL-encoded POSTs and REST requests that 
61   * require building and parsing JSON or XML.  Convenient access to a few common
62   * authentication methods is also available.</p>
63   * 
64   * 
65   * <h3>Conventions</h3>
66   * <p>HTTPBuilder has properties for default headers, URI, contentType, etc.  
67   * All of these values are also assignable (and in many cases, in much finer 
68   * detail) from the {@link RequestConfigDelegate} as well.  In any cases where the value
69   * is not set on the delegate (from within a request closure,) the builder's 
70   * default value is used.  </p>
71   * 
72   * <p>For instance, any methods that do not take a <code>uri</code> parameter 
73   * assume you will set the <code>uri</code> property in the request closure or 
74   * use HTTPBuilder's assigned {@link #getUri() default URI}.</p>
75   * 
76   * 
77   * <h3>Response Parsing</h3>
78   * <p>By default, HTTPBuilder uses {@link ContentType#ANY} as the default 
79   * content-type.  This means the value of the request's <code>Accept</code> 
80   * header is <code>&#42;/*</code>, and the response parser is determined 
81   * based on the response <code>content-type</code> header. </p>
82   * 
83   * <p><strong>If</strong> any contentType is given (either in 
84   * {@link #setContentType(Object)} or as a request method parameter), the 
85   * builder will attempt to parse the response using that content-type, 
86   * regardless of what the server actually responds with.  </p>
87   * 
88   *  
89   * <h3>Examples:</h3>
90   * Perform an HTTP GET and print the response:
91   * <pre>
92   *   def http = new HTTPBuilder('http://www.google.com')
93   *   
94   *   http.get( path : '/search', 
95   *             contentType : TEXT,
96   *             query : [q:'Groovy'] ) { resp, reader ->
97   *     println "response status: ${resp.statusLine}"
98   *     println 'Response data: -----'
99   *     System.out << reader
100  *     println '\n--------------------'
101  *   }
102  * </pre>
103  *   
104  * Long form for other HTTP methods, and response-code-specific handlers.  
105  * This is roughly equivalent to the above example.
106  *   
107  * <pre>
108  *   def http = new HTTPBuilder('http://www.google.com/search?q=groovy')
109  *   
110  *   http.request( GET, TEXT ) { req ->
111  *   
112  *     // executed for all successful responses:
113  *     response.success = { resp, reader ->
114  *       println 'my response handler!'
115  *       assert resp.statusLine.statusCode == 200
116  *       println resp.statusLine
117  *       System.out << reader // print response stream
118  *     }
119  *     
120  *     // executed only if the response status code is 401:
121  *     response.'404' = { resp -> 
122  *       println 'not found!'
123  *     }
124  *   }
125  * </pre>
126  *   
127  * You can also set a default response handler called for any status
128  * code > 399 that is not matched to a specific handler. Setting the value
129  * outside a request closure means it will apply to all future requests with
130  * this HTTPBuilder instance:
131  * <pre>
132  *   http.handler.failure = { resp ->
133  *     println "Unexpected failure: ${resp.statusLine}"
134  *   }
135  * </pre>
136  *   
137  *   
138  * And...  Automatic response parsing for registered content types!
139  *   
140  * <pre>
141  *   http.request( 'http://ajax.googleapis.com', GET, JSON ) {
142  *     uri.path = '/ajax/services/search/web'
143  *     uri.query = [ v:'1.0', q: 'Calvin and Hobbes' ]
144  *     
145  *     response.success = { resp, json ->
146  *       assert json.size() == 3
147  *       println "Query response: "
148  *       json.responseData.results.each {
149  *         println "  ${it.titleNoFormatting} : ${it.visibleUrl}"
150  *       }
151  *     }
152  *   }
153  * </pre>
154  * 
155  * 
156  * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
157  */

333 		return this.doRequest( delegate );
334 	}

394 	/**
395 	 * Create a {@link RequestConfigDelegate} from the given arguments, execute the 
396 	 * config closure, then pass the delegate to {@link #doRequest(RequestConfigDelegate)},
397 	 * which actually executes the request.
398 	 */
399 	protected Object doRequest( URI uri, Method method, Object contentType, Closure configClosure ) 
400 			throws ClientProtocolException, IOException {
401 
402 		HttpRequestBase reqMethod;
403 		try { reqMethod = method.getRequestType().newInstance();
404 		// this exception should reasonably never occur:
405 		} catch ( Exception e ) { throw new RuntimeException( e ); }
406 
407 		reqMethod.setURI( uri );
408 		RequestConfigDelegate delegate = new RequestConfigDelegate( reqMethod, contentType, 
409 				this.defaultRequestHeaders,
410 				this.defaultResponseHandlers );		
411 		configClosure.setDelegate( delegate );
412 		configClosure.setResolveStrategy( Closure.DELEGATE_FIRST );
413 		configClosure.call( reqMethod );
414 		
415 		return this.doRequest( delegate );
416 	}

418 	/**
419 	 * All <code>request</code> methods delegate to this method.
420 	 */
421 	protected Object doRequest( final RequestConfigDelegate delegate ) 
422 			throws ClientProtocolException, IOException {
423 		
424 		final HttpRequestBase reqMethod = delegate.getRequest();
425 		
426 		Object contentType = delegate.getContentType();
427 		String acceptContentTypes = contentType.toString();
428 		if ( contentType instanceof ContentType ) 
429 			acceptContentTypes = ((ContentType)contentType).getAcceptHeader();
430 		
431 		reqMethod.setHeader( "Accept", acceptContentTypes );
432 		reqMethod.setURI( delegate.getUri().toURI() );
433 		log.debug( reqMethod.getMethod() + " " + reqMethod.getURI() );
434 
435 		// set any request headers from the delegate
436 		Map<?,?> headers = delegate.getHeaders(); 
437 		for ( Object key : headers.keySet() ) {
438 			Object val = headers.get( key );
439 			if ( key == null ) continue;
440 			if ( val == null ) reqMethod.removeHeaders( key.toString() ); 
441 			else reqMethod.setHeader( key.toString(), val.toString() );
442 		}
443 		
444 		HttpResponse resp = client.execute( reqMethod );
445 		try {
446 			resp = new HttpResponseDecorator( resp, null );
447 			int status = resp.getStatusLine().getStatusCode();
448 			Closure responseClosure = delegate.findResponseHandler( status );
449 			log.debug( "Response code: " + status + "; found handler: " + responseClosure );
450 			
451 			Object[] closureArgs = null;
452 			switch ( responseClosure.getMaximumNumberOfParameters() ) {
453 			case 1 :
454 				closureArgs = new Object[] { resp };
455 				break;
456 			case 2 : // parse the response entity if the response handler expects it:
457 				try {
458 					closureArgs = new Object[] { resp, parseResponse( resp, contentType ) };
459 				}
460 				catch ( Exception ex ) {
461 					String respContentType = resp.getEntity().getContentType().getValue();
462 					log.warn( "Error parsing '" + respContentType + "' response", ex );
463 					throw new ResponseParseException( new HttpResponseDecorator( resp, null ) );	
464 				}
465 				break;
466 			default:
467 				throw new IllegalArgumentException( 
468 						"Response closure must accept one or two parameters" );
469 			}
470 			
471 			Object returnVal = responseClosure.call( closureArgs );
472 			log.trace( "response handler result: " + returnVal );
473 			
474 			return returnVal;
475 		}
476 		finally {
477 			HttpEntity entity = resp.getEntity(); 
478 			if ( entity != null ) entity.consumeContent();
479 		}
480 	}

482 	/**
483 	 * Parse the response data based on the given content-type.  
484 	 * If the given content-type is {@link ContentType#ANY}, the 
485 	 * <code>content-type</code> header from the response will be used to 
486 	 * determine how to parse the response. 
487 	 * @param resp
488 	 * @param contentType
489 	 * @return whatever was returned from the parser retrieved for the given 
490 	 *  content-type, or <code>null</code> if no parser could be found for this 
491 	 *  content-type.  The parser will also return <code>null</code> if the 
492 	 *  response does not contain any content (e.g. in response to a HEAD request).
493 	 * @throws HttpResponseException if there is a error parsing the response
494 	 */
495 	protected Object parseResponse( HttpResponse resp, Object contentType ) 
496 			throws HttpResponseException {
497 		// For HEAD or OPTIONS requests, there should be no response entity.
498 		if ( resp.getEntity() == null ) {
499 			log.debug( "Response contains no entity.  Parsed data is null." );
500 			return null;
501 		}
502 		// first, start with the _given_ content-type
503 		String responseContentType = contentType.toString();
504 		// if the given content-type is ANY ("*/*") then use the response content-type
505 		if ( ContentType.ANY.toString().equals( responseContentType ) )
506 			responseContentType = ParserRegistry.getContentType( resp );
507 		
508 		Object parsedData = null;
509 		Closure parser = parsers.getAt( responseContentType );
510 		if ( parser == null ) log.warn( "No parser found for content-type: " 
511 			+ responseContentType );
512 		else {
513 			log.debug( "Parsing response as: " + responseContentType );
514 			parsedData = parser.call( resp );
515 			if ( parsedData == null ) log.warn( "Parser returned null!" );
516 			else log.debug( "Parsed data to instance of: " + parsedData.getClass() );
517 		}
518 		return parsedData;
519 	}

533 		map.put(  Status.FAILURE,

536 		return map;
537 	}

539 	/**
540 	 * <p>This is the default <code>response.success</code> handler.  It will be 
541 	 * executed if the response is not handled by a status-code-specific handler 
542 	 * (i.e. <code>response.'200'= {..}</code>) and no generic 'success' handler 
543 	 * is given (i.e. <code>response.success = {..}</code>.)  This handler simply 
544 	 * returns the parsed data from the response body.  In most cases you will 
545 	 * probably want to define a <code>response.success = {...}</code> handler 
546 	 * from the request closure, which will replace the response handler defined 
547 	 * by this method.  </p>
548 	 * 
549 	 * <h4>Note for parsers that return streaming content:</h4>
550 	 * <p>For responses parsed as {@link ParserRegistry#parseStream(HttpResponse) 
551 	 * BINARY} or {@link ParserRegistry#parseText(HttpResponse) TEXT}, the 
552 	 * parser will return streaming content -- an <code>InputStream</code> or 
553 	 * <code>Reader</code>.  In these cases, this handler will buffer the the 
554 	 * response content before the network connection is closed.  </p>
555 	 * 
556 	 * <p>In practice, a user-supplied response handler closure is 
557 	 * <i>designed</i> to handle streaming content so it can be read directly from 
558 	 * the response stream without buffering, which will be much more efficient.
559 	 * Therefore, it is recommended that request method variants be used which 
560 	 * explicitly accept a response handler closure in these cases.</p>
561 	 *  
562 	 * @param resp HTTP response
563 	 * @param parsedData parsed data as resolved from this instance's {@link ParserRegistry}
564 	 * @return the parsed data object (whatever the parser returns).
565 	 */
566 	protected Object defaultSuccessHandler( HttpResponse resp, Object parsedData ) throws IOException {
567 		//If response is streaming, buffer it in a byte array:
568 		if ( parsedData instanceof InputStream ) {
569 			ByteArrayOutputStream buffer = new ByteArrayOutputStream();
570 			DefaultGroovyMethods.leftShift( buffer, (InputStream)parsedData );
571 			parsedData = new ByteArrayInputStream( buffer.toByteArray() );
572 		}
573 		else if ( parsedData instanceof Reader ) {
574 			StringWriter buffer = new StringWriter();
575 			DefaultGroovyMethods.leftShift( buffer, (Reader)parsedData );
576 			parsedData = new StringReader( buffer.toString() );
577 		}
578 		else if ( parsedData instanceof Closeable )
