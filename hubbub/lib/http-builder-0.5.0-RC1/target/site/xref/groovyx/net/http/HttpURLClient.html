52  /**
53   * <p>This class provides a simplified API similar to {@link HTTPBuilder}, but 
54   * uses {@link java.net.HttpURLConnection} for I/O so that it is compatible 
55   * with Google App Engine.  Features:
56   * <ul>
57   *  <li>Parser and Encoder support</li>
58   *  <li>Easy request and response header manipulation</li>
59   *  <li>Basic authentication</li>
60   * </ul>
61   * Notably absent are status-code based response handling and the more complex 
62   * authentication mechanisms.</p>
63   * 
64   * TODO request encoding support?
65   *
66   * @see http://code.google.com/appengine/docs/java/urlfetch/overview.html
67   * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
68   * @since 0.5.0
69   */

82  	/**
83  	 * Perform a request.  Parameters are:
84  	 * <dl>
85  	 *   <dt>url</dt><dd>the entire request URL</dd>
86  	 *   <dt>path</dt><dd>the path portion of the request URL, if a default
87  	 *     URL is set on this instance.</dd>
88  	 *   <dt>query</dt><dd>URL query parameters for this request.</dd>
89  	 *   <dt>timeout</dt><dd>see {@link HttpURLConnection#setReadTimeout(int)}</dd>
90  	 *   <dt>method</dt><dd>This defaults to GET, or POST if a <code>body</code> 
91  	 *   parameter is also specified.</dd>
92  	 *   <dt>contentType</dt><dd>Explicitly specify how to parse the response.
93  	 *     If this value is ContentType.ANY, the response <code>Content-Type</code>
94  	 *     header is used to determine how to parse the response.</dd>
95  	 *   <dt>requestContentType</dt><dd>used in a PUT or POST request to 
96  	 *     transform the request body and set the proper 
97  	 *     <code>Content-Type</code> header.  This defaults to the 
98  	 *     <code>contentType</code> if unset.</dd>
99  	 *   <dt>auth</dt><dd>Basic authorization; pass the value as a list in the 
100 	 *   form [user, pass]</dd>
101 	 *   <dt>headers</dt><dd>additional request headers, as a map</dd>
102 	 *   <dt>body</dt><dd>request content body, for a PUT or POST request.  
103 	 *     This will be encoded using the requestContentType</dd>
104 	 * </dl>
105 	 * @param args named parameters
106 	 * @return the parsed response
107 	 * @throws URISyntaxException
108 	 * @throws MalformedURLException
109 	 * @throws IOException
110 	 */
111 	public HttpResponseDecorator request( Map<String,?> args ) 
112 			throws URISyntaxException, MalformedURLException, IOException {
113 		
114 		// copy so we don't modify the original collection when removing items:
115 		args = new HashMap<String,Object>(args);
116 		
117 		Object arg = args.remove( "url" );
118 		if ( arg == null && this.defaultURL == null )
119 			throw new IllegalStateException( "Either the 'defaultURL' property" +
120 					" must be set or a 'url' parameter must be passed to the " +
121 					"request method." );
122 		URIBuilder url = arg != null ? new URIBuilder( arg.toString() ) : defaultURL.clone();
123 
124 		arg = null;
125 		arg = args.remove( "path" );
126 		if ( arg != null ) url.setPath( arg.toString() );
127 		arg = null;
128 		arg = args.remove( "query" );
129 		if ( arg != null ) {
130 			if ( ! ( arg instanceof Map ) ) 
131 				throw new IllegalArgumentException( "'query' must be a map" ); 
132 			url.setQuery( (Map<?,?>)arg );
133 		}
134 		
135 		HttpURLConnection conn = (HttpURLConnection)url.toURL().openConnection();
136 		conn.setInstanceFollowRedirects( this.followRedirects );
137 		
138 		arg = null;
139 		arg = args.remove( "timeout" );
140 		if ( arg != null ) 
141 			conn.setConnectTimeout( Integer.parseInt( arg.toString() ) );
142 		
143 		arg = null;
144 		arg = args.remove( "method" );
145 		if ( arg != null ) conn.setRequestMethod( arg.toString() );
146 
147 		arg = null;
148 		arg = args.remove( "contentType" );
149 		Object contentType = arg != null ? arg : this.contentType;
150 		if ( contentType instanceof ContentType ) conn.addRequestProperty( 
151 				"Accept", ((ContentType)contentType).getAcceptHeader() );
152 		
153 		arg = null;
154 		arg = args.remove( "requestContentType" );
155 		String requestContentType = arg != null ? arg.toString() : 	
156 				this.requestContentType != null ? this.requestContentType.toString() : 
157 					contentType != null ? contentType.toString() : null;
158 					
159 		// must add default headers before setting auth:
160 		for ( String key : defaultHeaders.keySet() )
161 			conn.addRequestProperty( key, defaultHeaders.get( key ) );
162 		
163 		arg = null;
164 		arg = args.remove( "auth" );
165 		if ( arg != null ) {
166 			try { 
167 				List<?> vals = (List<?>)arg;
168 				conn.addRequestProperty( "Authorization", getBasicAuthHeader( 
169 						vals.get(0).toString(), vals.get(1).toString() ) );
170 			} catch ( Exception ex ) {
171 				throw new IllegalArgumentException( 
172 						"Auth argument must be a list in the form [user,pass]" );
173 			}
174 		}
175 					
176 		arg = null;
177 		arg = args.remove( "headers" );
178 		if ( arg != null ) {
179 			if ( ! ( arg instanceof Map ) ) 
180 				throw new IllegalArgumentException( "'headers' must be a map" ); 
181 			Map<?,?> headers = (Map<?,?>)arg;
182 			for ( Object key : headers.keySet() ) conn.addRequestProperty( 
183 					key.toString(), headers.get( key ).toString() );
184 		}
185 		
186 		arg = null;
187 		arg = args.remove( "body" );
188 		if ( arg != null ) {
189 			conn.setDoOutput( true );
190 			HttpEntity body = (HttpEntity)encoderRegistry.getAt( 
191 					requestContentType ).call( arg );
192 			// TODO configurable request charset
193 			
194 			//TODO don't override if there is a 'content-type' in the headers list
195 			conn.addRequestProperty( "Content-Type", requestContentType );
196 			try {
197 				DefaultGroovyMethods.leftShift( conn.getOutputStream(), 
198 						body.getContent() );
199 			}
200 			finally { conn.getOutputStream().close(); }
201 		}
202 		if ( args.size() > 0 ) for ( Object k : args.keySet() ) 
203 			log.warn( "request() : Unkown named parameter '" + k + "'" );
204 		
205 		log.debug( conn.getRequestMethod() + " " + url );
206 		
207 		HttpResponse response = new HttpURLResponseAdapter(conn);
208 		if ( ContentType.ANY.equals( contentType ) ) contentType = conn.getContentType();
209 
210 		String method = conn.getRequestMethod();
211 		Object result = method.equals( "HEAD" ) || method.equals( "OPTIONS" ) ?
212 				null : parserRegistry.getAt( contentType ).call( response );
213 		
214 		log.debug( response.getStatusLine() );
215 		HttpResponseDecorator decoratedResponse = new HttpResponseDecorator( response, result );
216 		
217 		if ( log.isTraceEnabled() ) {
218 			for ( Header h : decoratedResponse.getHeaders() )
219 				log.trace( " << " + h.getName() + " : " + h.getValue() );
220 		}
221 				
222 		if ( conn.getResponseCode() > 399 ) 
223 			throw new HttpResponseException( decoratedResponse );
224 		
225 		return decoratedResponse;
226 	}

228 	private String getBasicAuthHeader( String user, String pass ) throws UnsupportedEncodingException {
229 	  return "Basic " + DefaultGroovyMethods.encodeBase64( 
230 			  (user + ":" + pass).getBytes("ISO-8859-1") ).toString();
231 	}

234 	 * Set basic user and password authorization to be used for every request.
235 	 * Pass <code>null</code> to un-set authorization for this instance.
236 	 * @param user 
237 	 * @param pass
238 	 * @throws UnsupportedEncodingException

240 	public void setBasicAuth( Object user, Object pass ) throws UnsupportedEncodingException {
241 		if ( user == null ) this.defaultHeaders.remove( "Authorization" );
242 		else this.defaultHeaders.put( "Authorization", 
243 				getBasicAuthHeader( user.toString(), pass.toString() ) );
244 	}

352 		public Header[] getAllHeaders() {
353 			if ( this.headers != null ) return this.headers;
354 			List<Header> headers = new ArrayList<Header>();
355 
356 			// see http://java.sun.com/j2se/1.5.0/docs/api/java/net/HttpURLConnection.html#getHeaderFieldKey(int)
357 			int i= conn.getHeaderFieldKey( 0 ) != null ? 0 : 1;
358 			String key;
359 			while ( ( key = conn.getHeaderFieldKey( i ) ) != null ) {
360 				headers.add( new BasicHeader( key, conn.getHeaderField( i++ ) ) );	
361 			}
362 
363 			this.headers = headers.toArray( new Header[headers.size()] ); 
364 			return this.headers;
365 		}

367 		public Header getFirstHeader( String key ) {
368 			for ( Header h : getAllHeaders() )

378 			List<Header> headers = new ArrayList<Header>();
379 			for ( Header h : getAllHeaders() )

381 			return headers.toArray( new Header[headers.size()] );
